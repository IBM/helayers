187 template <typename T>
188 void PrintUtils::printArray(std::ostream& out,
189  const std::vector<T>& vals,
190  bool withIndices)
191 {
192  const PrintOptions& po = getPrintOptions();
193  int num = po.numPrintedElements;
194  int num2 = po.numPrintedLastElements;
195  if (!withIndices)
196  out << "[";
197  for (int i = 0; i < std::min(num, (int)vals.size()); ++i) {
198  if (i > 0)
199  out << (withIndices ? " " : ",");
200  if (withIndices)
201  out << " [" << i << "]=";
202  out << vals[i];
203  }
204  int j = std::max((int)(vals.size() - num2), num);
205  if (j - num > 0)
206  out << " ... ";
207  for (; j < (int)vals.size(); ++j) {
208  if (j > 0)
209  out << " ";
210  if (withIndices)
211  out << " [" << j << "]=";
212  out << vals[j];
213  }
214  if (!withIndices)
215  out << "]";
216 }
