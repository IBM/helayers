211 template <typename T>
212 std::vector<T> MathUtils::getBatch(const std::vector<T>& vec,
213  size_t batchNum,
214  size_t batchSize)
215 {
216  size_t start = batchNum * batchSize;
217  if (start >= vec.size())
218  throw std::invalid_argument(
219  "Cannot get batch number " + std::to_string(batchNum) +
220  " with batch size " + std::to_string(batchSize) +
221  ", vector contains only " + std::to_string(vec.size()) + "elements");
222 
223  size_t end = std::min(start + batchSize, vec.size());
224  return std::vector<T>(vec.begin() + start, vec.begin() + end);
225 }
